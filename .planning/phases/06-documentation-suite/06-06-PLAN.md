import tempfile, pandas as pd
from pathlib import Path
from docx import Document
import sys
sys.path.insert(0, '.')
import src.doc_generator as dg
import src.config as cfg

EXPECTED_KEYS = {
    'analysis_md', 'analysis_docx', 'architecture', 'security',
    'engineering_decisions', 'data_handling', 'scalability',
    'system_design', 'alternatives'
}

with tempfile.TemporaryDirectory() as tmp:
    docs_dir = Path(tmp) / 'docs'
    df = pd.DataFrame({
        cfg.COL_RISK_LEVEL: ['CRITICAL']*45 + ['HIGH']*75 + ['MEDIUM']*90 + ['LOW']*90,
        cfg.COL_STUDENT_ID: [str(i) for i in range(300)],
    })
    run_log = {
        'run_timestamp': '2026-05-24T00:00:00+00:00',
        'students_processed': 300,
        'api_calls_made': 14,
        'tokens_used': {'input': 19693, 'output': 12078},
        'fallbacks_triggered': 0,
        'errors_encountered': [],
        'data_quality_warnings': [{'type': 'duplicate_id'}] * 9,
    }
    result = dg.write_docs(df, run_log, docs_dir)

    # 1. All 9 keys present
    missing_keys = EXPECTED_KEYS - set(result.keys())
    assert not missing_keys, f'Missing keys: {missing_keys}'

    # 2. All paths exist on disk
    for key, path in result.items():
        assert path.exists(), f'Path does not exist for key {key!r}: {path}'

    # 3. docs/ contains exactly 8 .docx files
    docx_files = list(docs_dir.glob('*.docx'))
    assert len(docx_files) == 8, f'Expected 8 .docx files, got {len(docx_files)}: {[f.name for f in docx_files]}'

    # 4. analysis.md at project root (docs_dir.parent)
    md_path = docs_dir.parent / 'analysis.md'
    assert md_path.exists(), f'analysis.md not found at {md_path}'
    md_text = md_path.read_text(encoding='utf-8')
    for heading in ['## Diagnosis', '## What You Found', '## What You Built', '## What You Cut', '## What Next']:
        assert heading in md_text, f'Missing heading: {heading}'
    word_count = len(md_text.split())
    assert word_count <= 600, f'analysis.md over 600 words: {word_count}'

    # 5. All .docx files open without error
    for f in docx_files:
        doc = Document(str(f))
        assert len(doc.paragraphs) > 0, f'{f.name} has no paragraphs'

    print(f'SMOKE TEST PASSED — 9 keys, 8 docx files, analysis.md {word_count} words')
"</automated>
